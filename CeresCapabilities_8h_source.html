43inline bool isLinearSolverAvailable(const std::string& solverName)
44{
45 if (solverName == "cuda-dense")
46 {
47 return hasCudaSupport();
48 }
49 else if (solverName == "cuda-sparse")
50 {
51 return hasCudaSupport() && hasSuiteSparseSupport();
52 }
53 else if (solverName == "sparse-schur")
54 {
55 // Sparse Schur works with EigenSparse even without SuiteSparse
56 return true;
57 }
58 // dense-qr, dense-schur, iterative-schur are always available
59 return true;
60}

65inline std::string getLinearSolverUnavailableReason(const std::string& solverName)
66{
67 if (solverName == "cuda-dense" && !hasCudaSupport())
68 {
69 return "Ceres was not built with CUDA support";
70 }
71 else if (solverName == "cuda-sparse")
72 {
73 if (!hasCudaSupport() && !hasSuiteSparseSupport())
74 {
75 return "Ceres was not built with CUDA or SuiteSparse support";
76 }
77 else if (!hasCudaSupport())
78 {
79 return "Ceres was not built with CUDA support";
80 }
81 else if (!hasSuiteSparseSupport())
82 {
83 return "Ceres was not built with SuiteSparse support";
84 }
85 }
86 return "";
87}
