24std::vector<int> parseGridIndices(const std::string& str)
25{
26 std::set<int> indices;
27 if (str.empty())
28 {
29 return {}; // Empty means load all
30 }
31
32 std::stringstream ss(str);
33 std::string token;
34
35 while (std::getline(ss, token, ','))
36 {
37 // Trim whitespace
38 size_t start = token.find_first_not_of(" \t");
39 size_t end = token.find_last_not_of(" \t");
40 if (start == std::string::npos)
41 continue;
42 token = token.substr(start, end - start + 1);
43
44 // Check for range (e.g., "0-5")
45 size_t dashPos = token.find('-');
46 if (dashPos != std::string::npos && dashPos > 0)
47 {
48 int rangeStart = std::stoi(token.substr(0, dashPos));
49 int rangeEnd = std::stoi(token.substr(dashPos + 1));
50 for (int i = rangeStart; i <= rangeEnd; ++i)
51 {
52 indices.insert(i);
53 }
54 }
55 else
56 {
57 indices.insert(std::stoi(token));
58 }
59 }
60
61 return std::vector<int>(indices.begin(), indices.end());
62}
