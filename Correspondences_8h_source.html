45inline std::vector<Correspondence> computeRayCorrespondences(
46 const Grid& srcGrid,
47 const Grid& tgtGrid,
48 const Eigen::Vector3f& rayDir,
49 const Eigen::Isometry3d& srcToTgt,
50 float maxDist,
51 int subsampleX = 1,
52 int subsampleY = 1)
53{
54 // Get ray origins from source grid (valid triangle vertices)
55 std::vector<int> srcIndices = srcGrid.getTriangleVertexIndices(subsampleX, subsampleY);
56 std::vector<float> origins = srcGrid.getTriangleVertices(subsampleX, subsampleY);
57 int nRays = static_cast<int>(origins.size() / 3);
58
59 if (nRays == 0)
60 {
61 return {};
62 }
63
64 // Use target grid's backend for intersection
65 auto hits = tgtGrid.getBackend().intersectParallel(
66 origins.data(),
67 nRays,
68 rayDir,
69 srcToTgt,
70 maxDist);
71
72 // Convert hits to correspondences
73 std::vector<Correspondence> correspondences;
74 correspondences.reserve(hits.size());
75
76 for (const auto& hit : hits)
77 {
78 Correspondence corr;
79 corr.srcVertexIdx = srcIndices[hit.rayIndex];
80 corr.srcPoint = srcGrid.getVertex(corr.srcVertexIdx);
81 corr.tgtPoint = hit.point;
82 corr.tgtNormal = hit.normal;
83 corr.tgtFacetId = hit.facetId;
84 corr.weight = 1.0f;
85
86 correspondences.push_back(corr);
87 }
88
89 return correspondences;
90}

93struct BidirectionalCorrs
94{

96 std::vector<Correspondence> reverse; // target → source

111inline BidirectionalCorrs computeBidirectionalCorrs(
112 const Grid& source,
113 const Grid& target,
114 const Eigen::Vector3f& rayDir,
115 const Eigen::Isometry3d& srcToTgt,
116 float maxDist,
117 int subsampleX = 1,
118 int subsampleY = 1)
119{
120 BidirectionalCorrs result;
121
122 // Forward: rays from source, intersect target
123 result.forward = computeRayCorrespondences(source, target, rayDir, srcToTgt, maxDist,
124 subsampleX, subsampleY);
125
126 // Reverse: rays from target, intersect source
127 Eigen::Isometry3d tgtToSrc = srcToTgt.inverse();
128 result.reverse = computeRayCorrespondences(target, source, rayDir, tgtToSrc, maxDist,
129 subsampleX, subsampleY);
130
131 return result;
132}
