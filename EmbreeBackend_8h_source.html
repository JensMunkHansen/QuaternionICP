42 void build(const Grid& grid) override
43 {
44 grid_ = &grid;
45
46 // Clean up any existing scene
47 if (scene_)
48 {
49 rtcReleaseScene(scene_);
50 scene_ = nullptr;
51 }
52 primIdToFacetId_.clear();
53
54 scene_ = rtcNewScene(device_);
55
56 // Extract triangles from grid
57 // Iterate over cells, each cell (quad) can have up to 2 triangles
58 const int nRows = grid.height;
59 const int nCols = grid.width;
60 const uint8_t* marks = grid.marksData();
61
62 // First pass: count triangles
63 int triCount = 0;
64 for (int j = 0; j < nRows - 1; j++)
65 {
66 for (int i = 0; i < nCols - 1; i++)
67 {
68 uint8_t m = marks[j * nCols + i];
69 if (m & 0x08) triCount++; // Upper triangle
70 if (m & 0x10) triCount++; // Lower triangle
71 }
72 }
73
74 if (triCount == 0)
75 {
76 rtcCommitScene(scene_);
77 return;
78 }
79
80 // Create geometry
81 RTCGeometry geom = rtcNewGeometry(device_, RTC_GEOMETRY_TYPE_TRIANGLE);
82
83 // Allocate vertex buffer - we'll store unique vertices per cell corner
84 // For simplicity, reuse the grid's vertex layout (nRows * nCols vertices)
85 float* vertices = (float*)rtcSetNewGeometryBuffer(
86 geom, RTC_BUFFER_TYPE_VERTEX, 0, RTC_FORMAT_FLOAT3,
87 3 * sizeof(float), nRows * nCols);
88
89 // Copy vertices from grid
90 std::memcpy(vertices, grid.verticesData(), nRows * nCols * 3 * sizeof(float));
91
92 // Allocate index buffer
93 unsigned* indices = (unsigned*)rtcSetNewGeometryBuffer(
94 geom, RTC_BUFFER_TYPE_INDEX, 0, RTC_FORMAT_UINT3,
95 3 * sizeof(unsigned), triCount);
96
97 // Reserve space for facetId mapping
98 primIdToFacetId_.reserve(triCount);
99
100 // Second pass: fill triangle indices
101 // Cell vertex layout (matching GridSearch):
102 // i0----i1 (row j)
103 // | |
104 // i3----i2 (row j+1)
105 //
106 // Mark bit 0x20 (MarkQuadDiagonal) determines diagonal:
107 // - Not set (diag v1-v3): Upper=UL(i0,i1,i3), Lower=LR(i1,i2,i3)
108 // - Set (diag v0-v2): Upper=UR(i0,i1,i2), Lower=LL(i0,i2,i3)
109 //
110 // FacetId encoding (for normal computation):
111 // - 0x30000 (UL): v0, v1, v3
112 // - 0x20000 (LR): v1, v2, v3
113 // - 0x10000 (UR): v0, v1, v2
114 // - 0x00000 (LL): v0, v2, v3
115 int primId = 0;
116 for (int j = 0; j < nRows - 1; j++)
117 {
118 for (int i = 0; i < nCols - 1; i++)
119 {
120 int i0 = j * nCols + i; // upper-left
121 int i1 = j * nCols + i + 1; // upper-right
122 int i2 = (j + 1) * nCols + i + 1; // lower-right
123 int i3 = (j + 1) * nCols + i; // lower-left
124
125 uint8_t m = marks[i0];
126 bool hasUpper = (m & 0x08) != 0;
127 bool hasLower = (m & 0x10) != 0;
128 bool diag02 = (m & 0x20) != 0;
129
130 if (hasUpper)
131 {
132 if (diag02)
133 {
134 // UR: i0, i1, i2
135 indices[primId * 3 + 0] = i0;
136 indices[primId * 3 + 1] = i1;
137 indices[primId * 3 + 2] = i2;
138 primIdToFacetId_.push_back(i0 | 0x10000); // UR config
139 }
140 else
141 {
142 // UL: i0, i1, i3
143 indices[primId * 3 + 0] = i0;
144 indices[primId * 3 + 1] = i1;
145 indices[primId * 3 + 2] = i3;
146 primIdToFacetId_.push_back(i0 | 0x30000); // UL config
147 }
148 primId++;
149 }
150
151 if (hasLower)
152 {
153 if (diag02)
154 {
155 // LL: i0, i2, i3
156 indices[primId * 3 + 0] = i0;
157 indices[primId * 3 + 1] = i2;
158 indices[primId * 3 + 2] = i3;
159 primIdToFacetId_.push_back(i0 | 0x00000); // LL config
160 }
161 else
162 {
163 // LR: i1, i2, i3
164 indices[primId * 3 + 0] = i1;
165 indices[primId * 3 + 1] = i2;
166 indices[primId * 3 + 2] = i3;
167 primIdToFacetId_.push_back(i0 | 0x20000); // LR config
168 }
169 primId++;
170 }
171 }
172 }
173
174 rtcCommitGeometry(geom);
175 rtcAttachGeometry(scene_, geom);
176 rtcReleaseGeometry(geom); // Scene holds reference now
177
178 rtcCommitScene(scene_);
179 }

181 std::vector<RayHit> intersectParallel(
182 const float* rayOrigins,
183 int numRays,
184 const Eigen::Vector3f& rayDir,
185 const Eigen::Isometry3d& rayToGrid,
186 float maxDist) const override
187 {
188 std::vector<RayHit> hits;
189
190 if (numRays == 0 || !scene_ || primIdToFacetId_.empty())
191 {
192 return hits;
193 }
194
195 hits.reserve(numRays / 2); // Assume ~50% hit rate
196
197 // Precompute transform components
198 Eigen::Matrix3f R = rayToGrid.rotation().cast<float>();
199 Eigen::Vector3f t = rayToGrid.translation().cast<float>();
200
201 // Transform ray direction once (direction is shared)
202 Eigen::Vector3f gridRayDir = R * rayDir;
203
204 // Process rays
205 for (int i = 0; i < numRays; i++)
206 {
207 // Get ray origin in source coordinates
208 Eigen::Vector3f srcOrigin(
209 rayOrigins[i * 3 + 0],
210 rayOrigins[i * 3 + 1],
211 rayOrigins[i * 3 + 2]);
212
213 // Transform to grid-local coordinates
214 Eigen::Vector3f gridOrigin = R * srcOrigin + t;
215
216 // Embree doesn't support negative tnear (intersections behind origin are ignored).
217 // For bidirectional search (matching GridSearch), shoot two rays in opposite directions
218 // and take the closest hit. This is the recommended approach per Intel:
219 // https://community.intel.com/t5/Intel-Embree-Ray-Tracing-Kernels/Bidirectional-Ray-Tracing/m-p/1564491
220
221 // Forward ray (along rayDir)
222 // Use small negative tnear to catch surface-on-surface hits at tâ‰ˆ0
223 constexpr float epsilon = 1e-4f;
224 RTCRayHit fwdRay;
225 fwdRay.ray.org_x = gridOrigin.x();
226 fwdRay.ray.org_y = gridOrigin.y();
227 fwdRay.ray.org_z = gridOrigin.z();
228 fwdRay.ray.dir_x = gridRayDir.x();
229 fwdRay.ray.dir_y = gridRayDir.y();
230 fwdRay.ray.dir_z = gridRayDir.z();
231 fwdRay.ray.tnear = -epsilon; // Allow hits slightly behind origin
232 fwdRay.ray.tfar = maxDist;
233 fwdRay.ray.mask = -1;
234 fwdRay.ray.flags = 0;
235 fwdRay.hit.geomID = RTC_INVALID_GEOMETRY_ID;
236 fwdRay.hit.primID = RTC_INVALID_GEOMETRY_ID;
237
238 // Backward ray (opposite direction)
239 RTCRayHit bwdRay;
240 bwdRay.ray.org_x = gridOrigin.x();
241 bwdRay.ray.org_y = gridOrigin.y();
242 bwdRay.ray.org_z = gridOrigin.z();
243 bwdRay.ray.dir_x = -gridRayDir.x();
244 bwdRay.ray.dir_y = -gridRayDir.y();
245 bwdRay.ray.dir_z = -gridRayDir.z();
246 bwdRay.ray.tnear = 0.0f;
247 bwdRay.ray.tfar = maxDist;
248 bwdRay.ray.mask = -1;
249 bwdRay.ray.flags = 0;
250 bwdRay.hit.geomID = RTC_INVALID_GEOMETRY_ID;
251 bwdRay.hit.primID = RTC_INVALID_GEOMETRY_ID;
252
253 // Trace both rays
254 rtcIntersect1(scene_, &fwdRay);
255 rtcIntersect1(scene_, &bwdRay);
256
257 // Take the closest hit from either direction
258 bool fwdHit = (fwdRay.hit.geomID != RTC_INVALID_GEOMETRY_ID);
259 bool bwdHit = (bwdRay.hit.geomID != RTC_INVALID_GEOMETRY_ID);
260
261 RTCRayHit* bestRay = nullptr;
262 Eigen::Vector3f hitDir = gridRayDir;
263
264 if (fwdHit && bwdHit)
265 {
266 // Both hit - take closer one
267 bestRay = (fwdRay.ray.tfar <= bwdRay.ray.tfar) ? &fwdRay : &bwdRay;
268 if (bestRay == &bwdRay) hitDir = -gridRayDir;
269 }
270 else if (fwdHit)
271 {
272 bestRay = &fwdRay;
273 }
274 else if (bwdHit)
275 {
276 bestRay = &bwdRay;
277 hitDir = -gridRayDir;
278 }
279
280 // Check for hit
281 if (bestRay)
282 {
283 RayHit hit;
284 hit.rayIndex = i;
285
286 // Compute hit point
287 hit.point = gridOrigin + hitDir * bestRay->ray.tfar;
288
289 // Look up facetId and compute normal using Grid's method
290 unsigned primID = bestRay->hit.primID;
291 if (primID < primIdToFacetId_.size())
292 {
293 hit.facetId = primIdToFacetId_[primID];
294 hit.normal = grid_->computeFacetNormalFromId(hit.facetId);
295 }
296 else
297 {
298 // Fallback: use Embree's geometric normal
299 hit.facetId = -1;
300 hit.normal = Eigen::Vector3f(bestRay->hit.Ng_x, bestRay->hit.Ng_y, bestRay->hit.Ng_z).normalized();
301 }
302
303 hits.push_back(hit);
304 }
305 }
306
307 return hits;
308 }
309
310private:
311 RTCDevice device_ = nullptr;
312 RTCScene scene_ = nullptr;
313 const Grid* grid_ = nullptr;
314
315 // Map from Embree geometry primID to Grid facetId
316 std::vector<int> primIdToFacetId_;
317};
