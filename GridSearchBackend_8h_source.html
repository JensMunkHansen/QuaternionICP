36 void build(const Grid& grid) override
37 {
38 grid_ = &grid;
39
40 // Store pointers to grid data (no copies)
41 vertices_ = grid.verticesData();
42 marks_ = grid.marksData();
43 nRows_ = grid.nRows();
44 nCols_ = grid.nCols();
45
46 // Create projection configuration
47 if (projection_)
48 {
49 GridSearch_DestroyProjection(projection_);
50 }
51 projection_ = GridSearch_CreateLinearProjectionWithOffset(
52 grid.dx(), grid.dy(),
53 grid.offsetX(), grid.offsetY(), grid.offsetZ());
54 }

56 std::vector<RayHit> intersectParallel(
57 const float* rayOrigins,
58 int numRays,
59 const Eigen::Vector3f& rayDir,
60 const Eigen::Isometry3d& rayToGrid,
61 float maxDist) const override
62 {
63 std::vector<RayHit> hits;
64
65 if (numRays == 0 || !projection_)
66 {
67 return hits;
68 }
69
70 // Row-major 4x4 transform for GridSearch
71 using RowMajor4d = Eigen::Matrix<double, 4, 4, Eigen::RowMajor>;
72 RowMajor4d worldToGrid = rayToGrid.matrix();
73
74 // Call GridSearch
75 GridSearchHit* results = nullptr;
76 int hitCount = 0;
77
78 GridSearch_ComputeIntersectionsParallel(
79 rayOrigins,
80 rayDir.data(),
81 numRays,
82 vertices_,
83 nRows_,
84 nCols_,
85 worldToGrid.data(),
86 marks_,
87 projection_,
88 &results,
89 &hitCount,
90 maxDist);
91
92 // Convert to RayHit with normals
93 hits.reserve(hitCount);
94 for (int h = 0; h < hitCount; h++)
95 {
96 const auto& gsHit = results[h];
97
98 RayHit hit;
99 hit.rayIndex = gsHit.rayIndex;
100 hit.facetId = gsHit.facetId;
101 hit.point = Eigen::Map<const Eigen::Vector3f>(gsHit.point);
102 hit.normal = grid_->computeFacetNormalFromId(gsHit.facetId);
103
104 hits.push_back(hit);
105 }
106
107 // Cleanup GridSearch results
108 GridSearch_FreeResults(results);
109
110 return hits;
111 }
112
113private:
114 GridSearchProjection* projection_ = nullptr;
115
116 // Pointers to Grid data (no copies, Grid must outlive backend)
117 const Grid* grid_ = nullptr;
118 const float* vertices_ = nullptr;
119 const uint8_t* marks_ = nullptr;
120 int nRows_ = 0;
121 int nCols_ = 0;
122};
