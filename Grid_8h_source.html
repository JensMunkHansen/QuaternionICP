83 Eigen::Vector3f computeFacetNormalFromId(int facetId) const
84 {
85 int i0 = facetId & 0xFFFF;
86 int config = facetId & 0x30000;
87
88 int i1 = i0 + 1;
89 int i2 = i0 + width + 1;
90 int i3 = i0 + width;
91
92 Eigen::Vector3f v0 = getVertex(i0);
93 Eigen::Vector3f v1 = getVertex(i1);
94 Eigen::Vector3f v2 = getVertex(i2);
95 Eigen::Vector3f v3 = getVertex(i3);
96
97 // Normal formulas from spsDecodeTriangulation25D (CCW winding, +Z for flat surface)
98 // v0=upper-left, v1=upper-right, v2=lower-right, v3=lower-left
99 Eigen::Vector3f normal;
100 switch (config)
101 {
102 case 0x30000: // UL (default diagonal, upper)
103 normal = (v3 - v1).cross(v0 - v1);
104 break;
105 case 0x20000: // LR (default diagonal, lower)
106 normal = (v3 - v2).cross(v1 - v2);
107 break;
108 case 0x10000: // UR (alternate diagonal, upper)
109 normal = (v2 - v1).cross(v0 - v1);
110 break;
111 default: // LL (alternate diagonal, lower, 0x00000)
112 normal = (v3 - v2).cross(v0 - v2);
113 break;
114 }
115 return normal.normalized();
116 }

118 // Test which side of diagonal the point is on (inline for performance)
119 // Returns true if point is in upper triangle
120 static inline bool isUpperTriangle(const Eigen::Vector3f& p,
121 const Eigen::Vector3f& v0, const Eigen::Vector3f& v1,
122 const Eigen::Vector3f& v2, const Eigen::Vector3f& v3, bool diag02)
123 {
124 if (diag02)
125 {
126 // Diagonal v0 -> v2: upper if point on left side
127 float sign = (p.x() - v2.x()) * (v0.y() - v2.y())
128 - (v0.x() - v2.x()) * (p.y() - v2.y());
129 return sign >= 0;
130 }
131 else
132 {
133 // Diagonal v1 -> v3: upper if point on left side
134 float sign = (p.x() - v3.x()) * (v1.y() - v3.y())
135 - (v1.x() - v3.x()) * (p.y() - v3.y());
136 return sign >= 0;
137 }
138 }

184 struct AABB
185 {
186 Eigen::Vector3f min = Eigen::Vector3f::Constant(std::numeric_limits<float>::max());
187 Eigen::Vector3f max = Eigen::Vector3f::Constant(std::numeric_limits<float>::lowest());
188
189 void expand(const Eigen::Vector3f& point)
190 {
191 min = min.cwiseMin(point);
192 max = max.cwiseMax(point);
193 }
194
195 bool overlaps(const AABB& other, float margin = 0.0f) const
196 {
197 return (min.x() - margin <= other.max.x()) && (max.x() + margin >= other.min.x()) &&
198 (min.y() - margin <= other.max.y()) && (max.y() + margin >= other.min.y()) &&
199 (min.z() - margin <= other.max.z()) && (max.z() + margin >= other.min.z());
200 }
201
202 Eigen::Vector3f center() const { return (min + max) * 0.5f; }
203 Eigen::Vector3f extents() const { return max - min; }
204 };
