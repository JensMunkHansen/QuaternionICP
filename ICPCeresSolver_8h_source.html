103template<typename JacobianPolicy = RayJacobianSimplified>
104CeresInnerSolveResult solveInnerCeres(
105 const std::vector<Correspondence>& fwdCorrs,
106 const std::vector<Correspondence>& revCorrs,
107 const Pose7& initialPose,
108 const Vector3& rayDir,
109 const GeometryWeighting& weighting = GeometryWeighting(),
110 const InnerParams& innerParams = InnerParams())
111{
112 CeresInnerSolveResult result;
113 result.iterations = 0;
114 result.converged = false;
115
116 // Convert Pose7 to Sophus::SE3d
117 Quaternion q0(initialPose[3], initialPose[0], initialPose[1], initialPose[2]);
118 Vector3 t0(initialPose[4], initialPose[5], initialPose[6]);
119 Sophus::SE3d pose(q0, t0);
120 double* pose_data = pose.data();
121
122 // Build Ceres problem
123 ceres::Problem problem;
124
125 // Set SE3 manifold for proper Lie group optimization
126 if (innerParams.useCustomManifold)
127 {
128 // Use our custom manifold with rotation scaling support
129 problem.AddParameterBlock(pose_data, 7, new SE3ScaledManifold(innerParams.rotationScale));
130 }
131 else
132 {
133 // Use Sophus manifold (ignores rotationScale, useful for comparison)
134 problem.AddParameterBlock(pose_data, Sophus::SE3d::num_parameters, new SophusSE3Manifold());
135 }
136
137 Vector3 rayDirD = rayDir.cast<double>();
138 double cosAngleThresh = weighting.enable_gate ? weighting.tau : 0.0;
139
140 // Add forward residuals
141 for (const auto& c : fwdCorrs)
142 {
143 double nDotD = c.tgtNormal.cast<double>().dot(rayDirD);
144 if (std::abs(nDotD) < cosAngleThresh)
145 continue;
146
147 problem.AddResidualBlock(
148 ForwardRayCost<JacobianPolicy>::Create(
149 c.srcPoint.cast<double>(),
150 c.tgtPoint.cast<double>(),
151 c.tgtNormal.cast<double>(),
152 rayDir,
153 weighting),
154 nullptr, pose_data);
155 }
156
157 // Add reverse residuals
158 for (const auto& c : revCorrs)
159 {
160 double nDotD = c.tgtNormal.cast<double>().dot(rayDirD);
161 if (std::abs(nDotD) < cosAngleThresh)
162 continue;
163
164 problem.AddResidualBlock(
165 ReverseRayCost<JacobianPolicy>::Create(
166 c.srcPoint.cast<double>(), // tgtPoint (q)
167 c.tgtPoint.cast<double>(), // srcPoint (p)
168 c.tgtNormal.cast<double>(), // srcNormal
169 rayDir,
170 weighting),
171 nullptr, pose_data);
172 }
173
174 result.valid_count = problem.NumResidualBlocks();
175
176 if (result.valid_count == 0)
177 {
178 result.pose = initialPose;
179 result.rms = std::numeric_limits<double>::max();
180 return result;
181 }
182
183 // Configure and solve
184 ceres::Solver::Options options = toCeresSolverOptions(innerParams);
185
186 ceres::Solve(options, &problem, &result.summary);
187
188 result.iterations = result.summary.iterations.size();
189 result.converged = (result.summary.termination_type == ceres::CONVERGENCE);
190 // Ceres uses 0.5 * sum(rÂ²), so multiply by 2 to get sum_sq for RMS
191 result.rms = std::sqrt(2.0 * result.summary.final_cost / result.valid_count);
192
193 // Convert Sophus::SE3d back to Pose7
194 Quaternion q_final = pose.unit_quaternion();
195 Vector3 t_final = pose.translation();
196
197 Pose7 finalPose;
198 finalPose << q_final.x(), q_final.y(), q_final.z(), q_final.w(),
199 t_final.x(), t_final.y(), t_final.z();
200
201 result.pose = finalPose;
202
203 if (innerParams.verbose)
204 {
205 std::cout << "\t\tCeres: " << result.iterations << " iterations, "
206 << "cost=" << 2.0 * result.summary.final_cost // Convert from Ceres's 0.5*sum to sum
207 << ", rms=" << result.rms << "\n";
208 }
209
210 return result;
211}

226template<typename JacobianPolicy = RayJacobianSimplified>
227CeresICPResult solveICPCeres(
228 const Grid& source,
229 const Grid& target,
230 const Pose7& initialPose,
231 const Vector3& rayDir,
232 const GeometryWeighting& weighting = GeometryWeighting(),
233 const InnerParams& innerParams = InnerParams(),
234 const OuterParams& outerParams = OuterParams())
235{
236 CeresICPResult result;
237 result.pose = initialPose;
238 result.outer_iterations = 0;
239 result.total_inner_iterations = 0;
240 result.converged = false;
241
242 Pose7 pose = initialPose;
243 double prev_rms = std::numeric_limits<double>::max();
244
245 for (int outer = 0; outer < outerParams.maxIterations; ++outer)
246 {
247 result.outer_iterations++;
248
249 // Compute correspondences at current pose
250 Eigen::Isometry3d srcToTgt = pose7ToIsometry(pose);
251 auto corrs = computeBidirectionalCorrs(
252 source, target, rayDir.cast<float>(), srcToTgt, outerParams.maxDist);
253
254 if (outerParams.verbose)
255 {
256 std::cout << "\touter " << outer << ":\n";
257 std::cout << "\t\tfwd_corrs=" << corrs.forward.size()
258 << ", rev_corrs=" << corrs.reverse.size() << "\n";
259 }
260
261 // Inner loop with Ceres
262 auto innerResult = solveInnerCeres<JacobianPolicy>(
263 corrs.forward, corrs.reverse, pose, rayDir, weighting, innerParams);
264
265 pose = innerResult.pose;
266 result.rms = innerResult.rms;
267 result.total_inner_iterations += innerResult.iterations;
268
269 if (outerParams.verbose)
270 {
271 std::cout << "\t\touter " << outer << ": rms=" << std::scientific
272 << std::setprecision(6) << result.rms << ", valid=" << innerResult.valid_count
273 << ", ceres_iters=" << innerResult.iterations << "\n";
274 }
275
276 // Check convergence based on RMS change across outer iterations
277 double rms_change = std::abs(prev_rms - result.rms);
278 if (result.rms < outerParams.convergenceTol ||
279 rms_change < outerParams.convergenceTol * result.rms)
280 {
281 result.converged = true;
282 break;
283 }
284 prev_rms = result.rms;
285 }
286
287 if (outerParams.verbose)
288 {
289 std::cout << "Total Ceres iterations: " << result.total_inner_iterations << "\n";
290 }
291
292 result.pose = pose;
293 return result;
294}
