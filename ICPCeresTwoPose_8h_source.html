82template<typename JacobianPolicy = RayJacobianSimplified>
83CeresTwoPoseInnerResult solveInnerCeresTwoPose(
84 const std::vector<Correspondence>& fwdCorrs,
85 const std::vector<Correspondence>& revCorrs,
86 const Pose7& initialPoseA,
87 const Pose7& initialPoseB,
88 const Vector3& rayDirA,
89 const Vector3& rayDirB,
90 const GeometryWeighting& weighting,
91 const ceres::Solver::Options& ceresOptions,
92 bool fixPoseA = false)
93{
94 CeresTwoPoseInnerResult result;
95 result.iterations = 0;
96 result.converged = false;
97
98 // Convert Pose7 to Sophus::SE3d for pose A
99 Quaternion qA0(initialPoseA[3], initialPoseA[0], initialPoseA[1], initialPoseA[2]);
100 Vector3 tA0(initialPoseA[4], initialPoseA[5], initialPoseA[6]);
101 Sophus::SE3d poseA(qA0, tA0);
102 double* poseA_data = poseA.data();
103
104 // Convert Pose7 to Sophus::SE3d for pose B
105 Quaternion qB0(initialPoseB[3], initialPoseB[0], initialPoseB[1], initialPoseB[2]);
106 Vector3 tB0(initialPoseB[4], initialPoseB[5], initialPoseB[6]);
107 Sophus::SE3d poseB(qB0, tB0);
108 double* poseB_data = poseB.data();
109
110 // Build Ceres problem
111 ceres::Problem problem;
112
113 // Set SE3 manifold for both poses
114 problem.AddParameterBlock(poseA_data, Sophus::SE3d::num_parameters, new SophusSE3Manifold());
115 problem.AddParameterBlock(poseB_data, Sophus::SE3d::num_parameters, new SophusSE3Manifold());
116
117 // Optionally fix pose A (useful for anchoring one grid)
118 if (fixPoseA)
119 {
120 problem.SetParameterBlockConstant(poseA_data);
121 }
122
123 double cosAngleThresh = weighting.enable_gate ? weighting.tau : 0.0;
124
125 // Add forward residuals (rays from A → B)
126 for (const auto& c : fwdCorrs)
127 {
128 // c.srcPoint is in frame A, c.tgtPoint/tgtNormal are in frame B
129 double nDotD = c.tgtNormal.cast<double>().dot(rayDirA);
130 if (std::abs(nDotD) < cosAngleThresh)
131 continue;
132
133 problem.AddResidualBlock(
134 ForwardRayCostTwoPose<JacobianPolicy>::Create(
135 c.srcPoint.cast<double>(), // pA: point in frame A
136 c.tgtPoint.cast<double>(), // qB: surface point in frame B
137 c.tgtNormal.cast<double>(), // nB: surface normal in frame B
138 rayDirA,
139 weighting),
140 nullptr, poseA_data, poseB_data);
141 }
142
143 // Add reverse residuals (rays from B → A)
144 for (const auto& c : revCorrs)
145 {
146 // For reverse: c.srcPoint is in frame B, c.tgtPoint/tgtNormal are in frame A
147 double nDotD = c.tgtNormal.cast<double>().dot(rayDirB);
148 if (std::abs(nDotD) < cosAngleThresh)
149 continue;
150
151 problem.AddResidualBlock(
152 ReverseRayCostTwoPose<JacobianPolicy>::Create(
153 c.srcPoint.cast<double>(), // pB: point in frame B
154 c.tgtPoint.cast<double>(), // qA: surface point in frame A
155 c.tgtNormal.cast<double>(), // nA: surface normal in frame A
156 rayDirB,
157 weighting),
158 nullptr, poseA_data, poseB_data);
159 }
160
161 result.valid_count = problem.NumResidualBlocks();
162
163 if (result.valid_count == 0)
164 {
165 result.poseA = initialPoseA;
166 result.poseB = initialPoseB;
167 result.rms = std::numeric_limits<double>::max();
168 return result;
169 }
170
171 // Solve
172 ceres::Solve(ceresOptions, &problem, &result.summary);
173
174 result.iterations = result.summary.iterations.size();
175 result.converged = (result.summary.termination_type == ceres::CONVERGENCE);
176 // Ceres uses 0.5 * sum(r²), so multiply by 2 to get sum_sq for RMS
177 result.rms = std::sqrt(2.0 * result.summary.final_cost / result.valid_count);
178
179 // Convert Sophus::SE3d back to Pose7 for pose A
180 Quaternion qA_final = poseA.unit_quaternion();
181 Vector3 tA_final = poseA.translation();
182 result.poseA << qA_final.x(), qA_final.y(), qA_final.z(), qA_final.w(),
183 tA_final.x(), tA_final.y(), tA_final.z();
184
185 // Convert Sophus::SE3d back to Pose7 for pose B
186 Quaternion qB_final = poseB.unit_quaternion();
187 Vector3 tB_final = poseB.translation();
188 result.poseB << qB_final.x(), qB_final.y(), qB_final.z(), qB_final.w(),
189 tB_final.x(), tB_final.y(), tB_final.z();
190
191 return result;
192}

198inline Pose7 computeRelativePose(const Pose7& poseA, const Pose7& poseB)
199{
200 // Convert to Isometry3d
201 Quaternion qA(poseA[3], poseA[0], poseA[1], poseA[2]);
202 Vector3 tA(poseA[4], poseA[5], poseA[6]);
203 Eigen::Isometry3d isoA = Eigen::Isometry3d::Identity();
204 isoA.linear() = qA.normalized().toRotationMatrix();
205 isoA.translation() = tA;
206
207 Quaternion qB(poseB[3], poseB[0], poseB[1], poseB[2]);
208 Vector3 tB(poseB[4], poseB[5], poseB[6]);
209 Eigen::Isometry3d isoB = Eigen::Isometry3d::Identity();
210 isoB.linear() = qB.normalized().toRotationMatrix();
211 isoB.translation() = tB;
212
213 // Compute srcToTgt: B^{-1} * A (transforms points from A to B frame)
214 Eigen::Isometry3d rel = isoB.inverse() * isoA;
215
216 // Convert back to Pose7
217 Quaternion qRel(rel.rotation());
218 qRel.normalize();
219 Pose7 result;
220 result << qRel.x(), qRel.y(), qRel.z(), qRel.w(),
221 rel.translation().x(), rel.translation().y(), rel.translation().z();
222 return result;
223}

239template<typename JacobianPolicy = RayJacobianSimplified>
240CeresTwoPoseResult solveICPCeresTwoPose(
241 const Grid& gridA,
242 const Grid& gridB,
243 const Pose7& initialPoseA,
244 const Pose7& initialPoseB,
245 const Vector3& rayDir,
246 const GeometryWeighting& weighting,
247 const InnerParams& innerParams,
248 const OuterParams& outerParams,
249 bool fixPoseA = false)
250{
251 CeresTwoPoseResult result;
252 result.poseA = initialPoseA;
253 result.poseB = initialPoseB;
254 result.outer_iterations = 0;
255 result.total_inner_iterations = 0;
256 result.converged = false;
257
258 // Convert InnerParams to Ceres options once
259 ceres::Solver::Options ceresOptions = toCeresSolverOptions(innerParams);
260
261 Pose7 poseA = initialPoseA;
262 Pose7 poseB = initialPoseB;
263 double prev_rms = std::numeric_limits<double>::max();
264
265 for (int outer = 0; outer < outerParams.maxIterations; ++outer)
266 {
267 result.outer_iterations++;
268
269 // Compute relative pose for correspondence computation: A_to_B = A * B^{-1}
270 Pose7 relPose = computeRelativePose(poseA, poseB);
271 Eigen::Isometry3d aToB = pose7ToIsometry(relPose);
272
273 // Compute bidirectional correspondences
274 // Forward: rays from A hitting B
275 // Reverse: rays from B hitting A
276 auto corrs = computeBidirectionalCorrs(
277 gridA, gridB, rayDir.cast<float>(), aToB, outerParams.maxDist,
278 outerParams.subsampleX, outerParams.subsampleY);
279
280 if (outerParams.verbose)
281 {
282 std::cout << "\touter " << outer << ":\n";
283 std::cout << "\t\tfwd_corrs=" << corrs.forward.size()
284 << ", rev_corrs=" << corrs.reverse.size() << "\n";
285 }
286
287 // Inner loop with Ceres two-pose solver
288 auto innerResult = solveInnerCeresTwoPose<JacobianPolicy>(
289 corrs.forward, corrs.reverse,
290 poseA, poseB,
291 rayDir, rayDir,
292 weighting, ceresOptions, fixPoseA);
293
294 poseA = innerResult.poseA;
295 poseB = innerResult.poseB;
296 result.rms = innerResult.rms;
297 result.total_inner_iterations += innerResult.iterations;
298
299 if (outerParams.verbose)
300 {
301 std::cout << "\t\touter " << outer << ": rms=" << std::scientific
302 << std::setprecision(6) << result.rms << ", valid=" << innerResult.valid_count
303 << ", ceres_iters=" << innerResult.iterations << "\n";
304 }
305
306 // Check convergence based on RMS change
307 double rms_change = std::abs(prev_rms - result.rms);
308 if (result.rms < outerParams.convergenceTol ||
309 rms_change < outerParams.convergenceTol * result.rms)
310 {
311 result.converged = true;
312 break;
313 }
314 prev_rms = result.rms;
315
316 // Early exit if Ceres converged in 1 iteration (same as MultiViewICP)
317 if (innerResult.iterations <= 1)
318 {
319 result.converged = true;
320 break;
321 }
322 }
323
324 if (outerParams.verbose)
325 {
326 std::cout << "Total Ceres TwoPose iterations: " << result.total_inner_iterations << "\n";
327 }
328
329 result.poseA = poseA;
330 result.poseB = poseB;
331 return result;
332}
