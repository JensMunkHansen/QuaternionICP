73template<typename JacobianPolicy = RayJacobianSimplified>
74double computeCostAtPose(
75 const std::vector<Correspondence>& fwdCorrs,
76 const std::vector<Correspondence>& revCorrs,
77 const Pose7& pose,
78 const Vector3& rayDir,
79 const GeometryWeighting& weighting)
80{
81 double cost = 0.0;
82
83 for (const auto& corr : fwdCorrs)
84 {
85 Vector3 pS = corr.srcPoint.cast<double>();
86 Vector3 qT = corr.tgtPoint.cast<double>();
87 Vector3 nT = corr.tgtNormal.cast<double>();
88 ForwardRayCost<JacobianPolicy> costFn(pS, qT, nT, rayDir, weighting);
89 double const* pparams[1] = {pose.data()};
90 double residual;
91 costFn(pparams, &residual, nullptr);
92 if (std::abs(residual) < 1e10) cost += residual * residual;
93 }
94
95 for (const auto& corr : revCorrs)
96 {
97 Vector3 pT = corr.srcPoint.cast<double>();
98 Vector3 qS = corr.tgtPoint.cast<double>();
99 Vector3 nS = corr.tgtNormal.cast<double>();
100 ReverseRayCost<JacobianPolicy> costFn(pT, qS, nS, rayDir, weighting);
101 double const* pparams[1] = {pose.data()};
102 double residual;
103 costFn(pparams, &residual, nullptr);
104 if (std::abs(residual) < 1e10) cost += residual * residual;
105 }
106
107 return cost;
108}

121inline bool isDeltaConverged(const Tangent6& delta, double transThreshold, double rotThreshold)
122{
123 double transNorm = delta.head<3>().norm();
124 double rotNorm = delta.tail<3>().norm();
125 return transNorm < transThreshold && rotNorm < rotThreshold;

138inline void applyDeltaAndNormalize(Pose7& pose, const Tangent6& delta, double rotationScale = 1.0)
139{
140 if (rotationScale == 1.0)
141 {
142 pose = se3Plus(pose, delta);
143 }
144 else
145 {
146 pose = se3PlusScaled(pose, delta, rotationScale);
147 }
148 Quaternion q(pose[3], pose[0], pose[1], pose[2]);
149 q.normalize();
150 pose.head<4>() << q.x(), q.y(), q.z(), q.w();
151}

172template<typename JacobianPolicy = RayJacobianSimplified>
173double lineSearch(
174 const std::vector<Correspondence>& fwdCorrs,
175 const std::vector<Correspondence>& revCorrs,
176 const Pose7& pose,
177 const Tangent6& delta,
178 double currentCost,
179 const Vector3& rayDir,
180 const GeometryWeighting& weighting,
181 const LineSearchParams& params,
182 double rotationScale = 1.0)
183{
184 double alpha = params.alpha;
185
186 for (int i = 0; i < params.maxIterations; ++i)
187 {
188 // Trial step with scaled delta
189 Tangent6 scaledDelta = alpha * delta;
190 Pose7 pose_trial = pose;
191 applyDeltaAndNormalize(pose_trial, scaledDelta, rotationScale);
192
193 double trialCost = computeCostAtPose<JacobianPolicy>(
194 fwdCorrs, revCorrs, pose_trial, rayDir, weighting);
195
196 if (trialCost < currentCost)
197 {
198 return alpha;
199 }
200
201 alpha *= params.beta;
202 }
203
204 return alpha; // Return smallest tried alpha
205}

230template<typename JacobianPolicy = RayJacobianSimplified>
231double buildNormalEquations(
232 const std::vector<Correspondence>& fwdCorrs,
233 const std::vector<Correspondence>& revCorrs,
234 const Pose7& pose,
235 const Vector3& rayDir,
236 const GeometryWeighting& weighting,
237 Matrix6& H,
238 Vector6& b,
239 int& fwd_valid,
240 int& rev_valid,
241 double rotationScale = 1.0)
242{
243 H.setZero();
244 b.setZero();
245 double sum_sq = 0.0;
246 fwd_valid = 0;
247 rev_valid = 0;
248
249 Matrix7x6 P = (rotationScale == 1.0)
250 ? plusJacobian7x6(pose)
251 : plusJacobian7x6Scaled(pose, rotationScale);
252
253 // Process forward correspondences (source→target)
254 for (const auto& corr : fwdCorrs)
255 {
256 Vector3 pS = corr.srcPoint.cast<double>();
257 Vector3 qT = corr.tgtPoint.cast<double>();
258 Vector3 nT = corr.tgtNormal.cast<double>();
259
260 ForwardRayCost<JacobianPolicy> cost(pS, qT, nT, rayDir, weighting);
261
262 double const* pparams[1] = {pose.data()};
263 double residual;
264 double J7[7];
265 double* jacobians[1] = {J7};
266
267 cost(pparams, &residual, jacobians);
268
269 if (std::abs(residual) > 1e10) continue;
270
271 Eigen::Map<Eigen::RowVectorXd> J7_map(J7, 7);
272 Eigen::Matrix<double, 1, 6> J6 = J7_map * P;
273
274 H += J6.transpose() * J6;
275 b += J6.transpose() * residual;
276 sum_sq += residual * residual;
277 fwd_valid++;
278 }
279
280 // Process reverse correspondences (target→source)
281 for (const auto& corr : revCorrs)
282 {
283 Vector3 pT = corr.srcPoint.cast<double>();
284 Vector3 qS = corr.tgtPoint.cast<double>();
285 Vector3 nS = corr.tgtNormal.cast<double>();
286
287 ReverseRayCost<JacobianPolicy> cost(pT, qS, nS, rayDir, weighting);
288
289 double const* pparams[1] = {pose.data()};
290 double residual;
291 double J7[7];
292 double* jacobians[1] = {J7};
293
294 cost(pparams, &residual, jacobians);
295
296 if (std::abs(residual) > 1e10) continue;
297
298 Eigen::Map<Eigen::RowVectorXd> J7_map(J7, 7);
299 Eigen::Matrix<double, 1, 6> J6 = J7_map * P;
300
301 H += J6.transpose() * J6;
302 b += J6.transpose() * residual;
303 sum_sq += residual * residual;
304 rev_valid++;
305 }
306
307 return sum_sq;
308}
