73template<typename JacobianPolicy = RayJacobianSimplified>
74double computeCostAtPose(
75 const std::vector<Correspondence>& fwdCorrs,
76 const std::vector<Correspondence>& revCorrs,
77 const Pose7& pose,
78 const Vector3& rayDir,
79 const GeometryWeighting& weighting)
80{
81 double cost = 0.0;
82
83 for (const auto& corr : fwdCorrs)
84 {
85 Vector3 pS = corr.srcPoint.cast<double>();
86 Vector3 qT = corr.tgtPoint.cast<double>();
87 Vector3 nT = corr.tgtNormal.cast<double>();
88 ForwardRayCost<JacobianPolicy> costFn(pS, qT, nT, rayDir, weighting);
89 double const* pparams[1] = {pose.data()};
90 double residual;
91 costFn(pparams, &residual, nullptr);
92 if (std::abs(residual) < 1e10) cost += residual * residual;
93 }
94
95 for (const auto& corr : revCorrs)
96 {
97 Vector3 pT = corr.srcPoint.cast<double>();
98 Vector3 qS = corr.tgtPoint.cast<double>();
99 Vector3 nS = corr.tgtNormal.cast<double>();
100 ReverseRayCost<JacobianPolicy> costFn(pT, qS, nS, rayDir, weighting);
101 double const* pparams[1] = {pose.data()};
102 double residual;
103 costFn(pparams, &residual, nullptr);
104 if (std::abs(residual) < 1e10) cost += residual * residual;
105 }
106
107 return cost;
108}

121inline bool isDeltaConverged(const Tangent6& delta, double transThreshold, double rotThreshold)
122{
123 double transNorm = delta.head<3>().norm();
124 double rotNorm = delta.tail<3>().norm();
125 return transNorm < transThreshold && rotNorm < rotThreshold;

137inline void applyDeltaAndNormalize(Pose7& pose, const Tangent6& delta)
138{
139 pose = se3Plus(pose, delta);
140 Quaternion q(pose[3], pose[0], pose[1], pose[2]);
141 q.normalize();
142 pose.head<4>() << q.x(), q.y(), q.z(), q.w();
143}

163template<typename JacobianPolicy = RayJacobianSimplified>
164double lineSearch(
165 const std::vector<Correspondence>& fwdCorrs,
166 const std::vector<Correspondence>& revCorrs,
167 const Pose7& pose,
168 const Tangent6& delta,
169 double currentCost,
170 const Vector3& rayDir,
171 const GeometryWeighting& weighting,
172 const LineSearchParams& params)
173{
174 double alpha = params.alpha;
175
176 for (int i = 0; i < params.maxIterations; ++i)
177 {
178 // Trial step with scaled delta
179 Tangent6 scaledDelta = alpha * delta;
180 Pose7 pose_trial = pose;
181 applyDeltaAndNormalize(pose_trial, scaledDelta);
182
183 double trialCost = computeCostAtPose<JacobianPolicy>(
184 fwdCorrs, revCorrs, pose_trial, rayDir, weighting);
185
186 if (trialCost < currentCost)
187 {
188 return alpha;
189 }
190
191 alpha *= params.beta;
192 }
193
194 return alpha; // Return smallest tried alpha
195}

219template<typename JacobianPolicy = RayJacobianSimplified>
220double buildNormalEquations(
221 const std::vector<Correspondence>& fwdCorrs,
222 const std::vector<Correspondence>& revCorrs,
223 const Pose7& pose,
224 const Vector3& rayDir,
225 const GeometryWeighting& weighting,
226 Matrix6& H,
227 Vector6& b,
228 int& fwd_valid,
229 int& rev_valid)
230{
231 H.setZero();
232 b.setZero();
233 double sum_sq = 0.0;
234 fwd_valid = 0;
235 rev_valid = 0;
236
237 Matrix7x6 P = plusJacobian7x6(pose);
238
239 // Process forward correspondences (source→target)
240 for (const auto& corr : fwdCorrs)
241 {
242 Vector3 pS = corr.srcPoint.cast<double>();
243 Vector3 qT = corr.tgtPoint.cast<double>();
244 Vector3 nT = corr.tgtNormal.cast<double>();
245
246 ForwardRayCost<JacobianPolicy> cost(pS, qT, nT, rayDir, weighting);
247
248 double const* pparams[1] = {pose.data()};
249 double residual;
250 double J7[7];
251 double* jacobians[1] = {J7};
252
253 cost(pparams, &residual, jacobians);
254
255 if (std::abs(residual) > 1e10) continue;
256
257 Eigen::Map<Eigen::RowVectorXd> J7_map(J7, 7);
258 Eigen::Matrix<double, 1, 6> J6 = J7_map * P;
259
260 H += J6.transpose() * J6;
261 b += J6.transpose() * residual;
262 sum_sq += residual * residual;
263 fwd_valid++;
264 }
265
266 // Process reverse correspondences (target→source)
267 for (const auto& corr : revCorrs)
268 {
269 Vector3 pT = corr.srcPoint.cast<double>();
270 Vector3 qS = corr.tgtPoint.cast<double>();
271 Vector3 nS = corr.tgtNormal.cast<double>();
272
273 ReverseRayCost<JacobianPolicy> cost(pT, qS, nS, rayDir, weighting);
274
275 double const* pparams[1] = {pose.data()};
276 double residual;
277 double J7[7];
278 double* jacobians[1] = {J7};
279
280 cost(pparams, &residual, jacobians);
281
282 if (std::abs(residual) > 1e10) continue;
283
284 Eigen::Map<Eigen::RowVectorXd> J7_map(J7, 7);
285 Eigen::Matrix<double, 1, 6> J6 = J7_map * P;
286
287 H += J6.transpose() * J6;
288 b += J6.transpose() * residual;
289 sum_sq += residual * residual;
290 rev_valid++;
291 }
292
293 return sum_sq;
294}
