142struct GeometryWeighting
143{
144 bool enable_weight = true;
145 bool enable_gate = true;
146 double tau = 0.2;
147 WeightingMode mode = WeightingMode::SqrtAbs;
148
154 double weight(double c) const
155 {
156 double ac = std::abs(c);
157 if (enable_gate && ac < tau)
158 return 0.0;
159 if (!enable_weight)
160 return 1.0;
161 ac = std::max(tau, std::min(1.0, ac));
162 return (mode == WeightingMode::SqrtAbs) ? std::sqrt(ac) : ac;
163 }
164};

379inline ceres::Solver::Options toCeresSolverOptions(const InnerParams& inner)
380{
381 ceres::Solver::Options options;
382
383 options.max_num_iterations = inner.maxIterations;
384 options.function_tolerance = inner.translationThreshold;
385 options.gradient_tolerance = inner.translationThreshold;
386 options.parameter_tolerance = inner.translationThreshold;
387
388 // Solver type
389 if (inner.solverType == SolverType::LevenbergMarquardt)
390 {
391 options.minimizer_type = ceres::TRUST_REGION;
392 options.trust_region_strategy_type = ceres::LEVENBERG_MARQUARDT;
393 options.initial_trust_region_radius = (inner.lm.lambda > 0) ? 1.0 / inner.lm.lambda : 1e4;
394 options.max_trust_region_radius = (inner.lm.lambdaMin > 0) ? 1.0 / inner.lm.lambdaMin : 1e8;
395 }
396 else
397 {
398 options.minimizer_type = ceres::TRUST_REGION;
399 options.trust_region_strategy_type = ceres::LEVENBERG_MARQUARDT;
400 options.initial_trust_region_radius = 1e16; // Very large = effectively GN
401 options.max_trust_region_radius = 1e32;
402 }
403
404 // Linear solver
405 switch (inner.linearSolverType)
406 {
407 case LinearSolverType::DenseQR:
408 options.linear_solver_type = ceres::DENSE_QR;
409 break;
410 case LinearSolverType::DenseSchur:
411 options.linear_solver_type = ceres::DENSE_SCHUR;
412 break;
413 case LinearSolverType::SparseSchur:
414 options.linear_solver_type = ceres::SPARSE_SCHUR;
415 break;
416 case LinearSolverType::IterativeSchur:
417 options.linear_solver_type = ceres::ITERATIVE_SCHUR;
418 break;
419 case LinearSolverType::CudaDenseCholesky:
420 options.linear_solver_type = ceres::DENSE_NORMAL_CHOLESKY;
421 options.dense_linear_algebra_library_type = ceres::CUDA;
422 break;
423 case LinearSolverType::CudaSparseCholesky:
424 // GPU-accelerated sparse via SuiteSparse CHOLMOD (requires CHOLMOD built with CUDA)
425 options.linear_solver_type = ceres::SPARSE_NORMAL_CHOLESKY;
426 options.sparse_linear_algebra_library_type = ceres::SUITE_SPARSE;
427 break;
428 }
429
430 // Preconditioner
431 switch (inner.preconditionerType)
432 {
433 case PreconditionerType::Identity:
434 options.preconditioner_type = ceres::IDENTITY;
435 break;
436 case PreconditionerType::Jacobi:
437 options.preconditioner_type = ceres::JACOBI;
438 break;
439 case PreconditionerType::SchurJacobi:
440 options.preconditioner_type = ceres::SCHUR_JACOBI;
441 break;
442 }
443
444 options.logging_type = inner.verbose ? ceres::PER_MINIMIZER_ITERATION : ceres::SILENT;
445 options.minimizer_progress_to_stdout = inner.verbose;
446
447 return options;
448}
