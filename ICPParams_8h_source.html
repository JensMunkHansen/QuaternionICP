142struct GeometryWeighting
143{
144 bool enable_weight = true;
145 bool enable_gate = true;
146 double tau = 0.2;
147 WeightingMode mode = WeightingMode::SqrtAbs;
148
154 double weight(double c) const
155 {
156 double ac = std::abs(c);
157 if (enable_gate && ac < tau)
158 return 0.0;
159 if (!enable_weight)
160 return 1.0;
161 ac = std::max(tau, std::min(1.0, ac));
162 return (mode == WeightingMode::SqrtAbs) ? std::sqrt(ac) : ac;
163 }
164};

386inline ceres::Solver::Options toCeresSolverOptions(const InnerParams& inner)
387{
388 ceres::Solver::Options options;
389
390 options.max_num_iterations = inner.maxIterations;
391 options.function_tolerance = inner.translationThreshold;
392 options.gradient_tolerance = inner.translationThreshold;
393 options.parameter_tolerance = inner.translationThreshold;
394
395 // Solver type
396 if (inner.solverType == SolverType::LevenbergMarquardt)
397 {
398 options.minimizer_type = ceres::TRUST_REGION;
399 options.trust_region_strategy_type = ceres::LEVENBERG_MARQUARDT;
400 options.initial_trust_region_radius = (inner.lm.lambda > 0) ? 1.0 / inner.lm.lambda : 1e4;
401 options.max_trust_region_radius = (inner.lm.lambdaMin > 0) ? 1.0 / inner.lm.lambdaMin : 1e8;
402 }
403 else
404 {
405 options.minimizer_type = ceres::TRUST_REGION;
406 options.trust_region_strategy_type = ceres::LEVENBERG_MARQUARDT;
407 options.initial_trust_region_radius = 1e16; // Very large = effectively GN
408 options.max_trust_region_radius = 1e32;
409 }
410
411 // Linear solver
412 switch (inner.linearSolverType)
413 {
414 case LinearSolverType::DenseQR:
415 options.linear_solver_type = ceres::DENSE_QR;
416 break;
417 case LinearSolverType::DenseSchur:
418 options.linear_solver_type = ceres::DENSE_SCHUR;
419 break;
420 case LinearSolverType::SparseSchur:
421 options.linear_solver_type = ceres::SPARSE_SCHUR;
422 break;
423 case LinearSolverType::IterativeSchur:
424 options.linear_solver_type = ceres::ITERATIVE_SCHUR;
425 break;
426 case LinearSolverType::CudaDenseCholesky:
427 options.linear_solver_type = ceres::DENSE_NORMAL_CHOLESKY;
428 options.dense_linear_algebra_library_type = ceres::CUDA;
429 break;
430 case LinearSolverType::CudaSparseCholesky:
431 // GPU-accelerated sparse via SuiteSparse CHOLMOD (requires CHOLMOD built with CUDA)
432 options.linear_solver_type = ceres::SPARSE_NORMAL_CHOLESKY;
433 options.sparse_linear_algebra_library_type = ceres::SUITE_SPARSE;
434 break;
435 }
436
437 // Preconditioner
438 switch (inner.preconditionerType)
439 {
440 case PreconditionerType::Identity:
441 options.preconditioner_type = ceres::IDENTITY;
442 break;
443 case PreconditionerType::Jacobi:
444 options.preconditioner_type = ceres::JACOBI;
445 break;
446 case PreconditionerType::SchurJacobi:
447 options.preconditioner_type = ceres::SCHUR_JACOBI;
448 break;
449 }
450
451 options.logging_type = inner.verbose ? ceres::PER_MINIMIZER_ITERATION : ceres::SILENT;
452 options.minimizer_progress_to_stdout = inner.verbose;
453
454 return options;
455}
