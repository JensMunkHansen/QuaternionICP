52inline Pose7 se3PlusLeft(const Pose7& x, const Tangent6& delta)
53{
54 // Extract quaternion and translation
55 Quaternion q(x[3], x[0], x[1], x[2]);
56 q.normalize();
57 Vector3 t = x.tail<3>();
58
59 // Extract tangent components
60 Vector3 v = delta.head<3>();
61 Vector3 w = delta.tail<3>();
62
63 // Left-multiplicative (space-frame) update:
64 // q_new = Exp(w) ⊗ q (left multiplication)
65 // t_new = deltaR * t + V(w) * v
66 Quaternion dq = quatExpSO3(w);
67 Quaternion q_new = (dq * q).normalized();
68
69 // Translation update for left multiplication
70 Matrix3 deltaR = dq.toRotationMatrix();
71 Matrix3 V = Vso3(w);
72 Vector3 t_new = deltaR * t + V * v;
73
74 // Pack result
75 Pose7 result;
76 result << q_new.x(), q_new.y(), q_new.z(), q_new.w(), t_new;
77 return result;
78}

83inline void applyDeltaLeftAndNormalize(Pose7& pose, const Tangent6& delta)
84{
85 pose = se3PlusLeft(pose, delta);
86 Quaternion q(pose[3], pose[0], pose[1], pose[2]);
87 q.normalize();
88 pose.head<4>() << q.x(), q.y(), q.z(), q.w();
89}

113template<typename JacobianPolicy = RayJacobianSimplified>
114double buildNormalEquationsLeft(
115 const std::vector<Correspondence>& fwdCorrs,
116 const std::vector<Correspondence>& revCorrs,
117 const Pose7& pose,
118 const Vector3& rayDirSource,
119 const GeometryWeighting& weighting,
120 Matrix6& H,
121 Vector6& b,
122 int& fwd_valid,
123 int& rev_valid)
124{
125 H.setZero();
126 b.setZero();
127 double sum_sq = 0.0;
128 fwd_valid = 0;
129 rev_valid = 0;
130
131 // Extract R and t from pose
132 Quaternion q(pose[3], pose[0], pose[1], pose[2]);
133 q.normalize();
134 Matrix3 R = q.toRotationMatrix();
135 Vector3 t = pose.tail<3>();
136
137 // Ray direction in target frame (for forward Jacobian)
138 Eigen::Vector3f rayDirTarget = (R * rayDirSource).cast<float>();
139 Eigen::Vector3f rayDirSrc = rayDirSource.cast<float>();
140
141 // Process forward correspondences (source→target)
142 // ForwardRayLeft expects ray direction in target frame
143 for (const auto& corr : fwdCorrs)
144 {
145 ForwardRayLeft<JacobianPolicy> cost(
146 corr.srcPoint, corr.tgtPoint, corr.tgtNormal, rayDirTarget, weighting);
147
148 Eigen::Matrix<double, 1, 6> J6;
149 double residual = cost.Evaluate(R, t, J6);
150
151 if (std::abs(residual) > 1e10) continue;
152
153 H += J6.transpose() * J6;
154 b += J6.transpose() * residual;
155 sum_sq += residual * residual;
156 fwd_valid++;
157 }
158
159 // Process reverse correspondences (target→source)
160 // ReverseRayLeft expects ray direction in source frame
161 for (const auto& corr : revCorrs)
162 {
163 ReverseRayLeft<JacobianPolicy> cost(
164 corr.srcPoint, corr.tgtPoint, corr.tgtNormal, rayDirSrc, weighting);
165
166 Eigen::Matrix<double, 1, 6> J6;
167 double residual = cost.Evaluate(R, t, J6);
168
169 if (std::abs(residual) > 1e10) continue;
170
171 H += J6.transpose() * J6;
172 b += J6.transpose() * residual;
173 sum_sq += residual * residual;
174 rev_valid++;
175 }
176
177 return sum_sq;
178}

183template<typename JacobianPolicy = RayJacobianSimplified>
184double computeCostAtPoseLeft(
185 const std::vector<Correspondence>& fwdCorrs,
186 const std::vector<Correspondence>& revCorrs,
187 const Pose7& pose,
188 const Vector3& rayDirSource,
189 const GeometryWeighting& weighting)
190{
191 double cost = 0.0;
192
193 // Extract R and t from pose
194 Quaternion q(pose[3], pose[0], pose[1], pose[2]);
195 q.normalize();
196 Matrix3 R = q.toRotationMatrix();
197 Vector3 t = pose.tail<3>();
198
199 Eigen::Vector3f rayDirTarget = (R * rayDirSource).cast<float>();
200 Eigen::Vector3f rayDirSrc = rayDirSource.cast<float>();
201
202 for (const auto& corr : fwdCorrs)
203 {
204 ForwardRayLeft<JacobianPolicy> costFn(
205 corr.srcPoint, corr.tgtPoint, corr.tgtNormal, rayDirTarget, weighting);
206 double residual = costFn.Residual(R, t);
207 if (std::abs(residual) < 1e10) cost += residual * residual;
208 }
209
210 for (const auto& corr : revCorrs)
211 {
212 ReverseRayLeft<JacobianPolicy> costFn(
213 corr.srcPoint, corr.tgtPoint, corr.tgtNormal, rayDirSrc, weighting);
214 double residual = costFn.Residual(R, t);
215 if (std::abs(residual) < 1e10) cost += residual * residual;
216 }
217
218 return cost;
219}

238template<typename JacobianPolicy = RayJacobianSimplified>
239InnerSolveResult solveInnerGNLeft(
240 const std::vector<Correspondence>& fwdCorrs,
241 const std::vector<Correspondence>& revCorrs,
242 const Pose7& initialPose,
243 const Vector3& rayDir,
244 const GeometryWeighting& weighting = GeometryWeighting(),
245 const InnerParams& params = InnerParams())
246{
247 InnerSolveResult result;
248 result.pose = initialPose;
249 result.iterations = 0;
250 result.converged = false;
251
252 Pose7 pose = initialPose;
253 const double lambda = params.damping;
254
255 for (int iter = 0; iter < params.maxIterations; ++iter)
256 {
257 result.iterations++;
258
259 // Build normal equations using left-perturbation Jacobians
260 Matrix6 H;
261 Vector6 b;
262 int fwd_valid, rev_valid;
263 double current_cost = buildNormalEquationsLeft<JacobianPolicy>(
264 fwdCorrs, revCorrs, pose, rayDir, weighting, H, b, fwd_valid, rev_valid);
265
266 int valid_count = fwd_valid + rev_valid;
267 result.rms = valid_count > 0 ? std::sqrt(current_cost / valid_count) : 0.0;
268 result.valid_count = valid_count;
269
270 if (params.verbose)
271 {
272 std::cout << "\t\t\tinner(left) " << iter << ": rms=" << std::scientific
273 << std::setprecision(6) << result.rms
274 << ", cost=" << current_cost
275 << ", fwd=" << fwd_valid << ", rev=" << rev_valid << "\n";
276 }
277
278 if (valid_count < 6)
279 {
280 result.pose = pose;
281 return result;
282 }
283
284 // Solve with damping: (H + lambda*I) delta = -b
285 H.diagonal().array() += lambda;
286 Tangent6 delta = -H.ldlt().solve(b);
287
288 // Apply step using LEFT multiplication
289 applyDeltaLeftAndNormalize(pose, delta);
290
291 // Check convergence
292 if (isDeltaConverged(delta, params.translationThreshold, params.rotationThreshold))
293 {
294 result.converged = true;
295 break;
296 }
297 }
298
299 result.pose = pose;
300 return result;
301}

306template<typename JacobianPolicy = RayJacobianSimplified>
307ICPResult solveICPLeft(
308 const Grid& source,
309 const Grid& target,
310 const Pose7& initialPose,
311 const Vector3& rayDir,
312 const GeometryWeighting& weighting = GeometryWeighting(),
313 const InnerParams& innerParams = InnerParams(),
314 const OuterParams& outerParams = OuterParams())
315{
316 ICPResult result;
317 result.pose = initialPose;
318 result.outer_iterations = 0;
319 result.total_inner_iterations = 0;
320 result.converged = false;
321
322 Pose7 pose = initialPose;
323 double prev_rms = std::numeric_limits<double>::max();
324
325 for (int outer = 0; outer < outerParams.maxIterations; ++outer)
326 {
327 result.outer_iterations++;
328
329 // Compute correspondences at current pose
330 Eigen::Isometry3d srcToTgt = pose7ToIsometry(pose);
331 auto corrs = computeBidirectionalCorrs(
332 source, target, rayDir.cast<float>(), srcToTgt, outerParams.maxDist);
333
334 if (outerParams.verbose)
335 {
336 std::cout << "\touter(left) " << outer << ":\n";
337 std::cout << "\t\tfwd_corrs=" << corrs.forward.size()
338 << ", rev_corrs=" << corrs.reverse.size() << "\n";
339 }
340
341 // Inner loop with fixed correspondences
342 auto innerResult = solveInnerGNLeft<JacobianPolicy>(
343 corrs.forward, corrs.reverse, pose, rayDir, weighting, innerParams);
344
345 pose = innerResult.pose;
346 result.rms = innerResult.rms;
347 result.total_inner_iterations += innerResult.iterations;
348
349 if (outerParams.verbose)
350 {
351 std::cout << "\t\touter(left) " << outer << ": rms=" << std::scientific
352 << std::setprecision(6) << result.rms << ", valid=" << innerResult.valid_count
353 << ", inner_iters=" << innerResult.iterations << "\n";
354 }
355
356 // Check convergence
357 double rms_change = std::abs(prev_rms - result.rms);
358 if (result.rms < outerParams.convergenceTol ||
359 rms_change < outerParams.convergenceTol * result.rms)
360 {
361 result.converged = true;
362 break;
363 }
364 prev_rms = result.rms;
365 }
366
367 if (outerParams.verbose)
368 {
369 std::cout << "Total inner iterations (left): " << result.total_inner_iterations << "\n";
370 }
371
372 result.pose = pose;
373 return result;
374}
