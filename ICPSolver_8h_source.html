53template<typename JacobianPolicy = RayJacobianSimplified>
54InnerSolveResult solveInnerGN(
55 const std::vector<Correspondence>& fwdCorrs,
56 const std::vector<Correspondence>& revCorrs,
57 const Pose7& initialPose,
58 const Vector3& rayDir,
59 const GeometryWeighting& weighting = GeometryWeighting(),
60 const InnerParams& params = InnerParams())
61{
62 InnerSolveResult result;
63 result.pose = initialPose;
64 result.iterations = 0;
65 result.converged = false;
66
67 Pose7 pose = initialPose;
68 const double lambda = params.damping;
69
70 const double rotationScale = params.rotationScale;
71
72 for (int iter = 0; iter < params.maxIterations; ++iter)
73 {
74 result.iterations++;
75
76 // Build normal equations and get current cost in one pass
77 Matrix6 H;
78 Vector6 b;
79 int fwd_valid, rev_valid;
80 double current_cost = buildNormalEquations<JacobianPolicy>(
81 fwdCorrs, revCorrs, pose, rayDir, weighting,
82 H, b, fwd_valid, rev_valid, rotationScale);
83
84 int valid_count = fwd_valid + rev_valid;
85 result.rms = valid_count > 0 ? std::sqrt(current_cost / valid_count) : 0.0;
86 result.valid_count = valid_count;
87
88 if (params.verbose)
89 {
90 std::cout << "\t\t\tinner " << iter << ": rms=" << std::scientific
91 << std::setprecision(6) << result.rms
92 << ", cost=" << current_cost
93 << ", fwd=" << fwd_valid << ", rev=" << rev_valid << "\n";
94 }
95
96 if (valid_count < 6)
97 {
98 result.pose = pose;
99 return result;
100 }
101
102 // Solve with damping: (H + lambda*I) delta = -b
103 H.diagonal().array() += lambda;
104 Tangent6 delta = -H.ldlt().solve(b);
105
106 // Line search if enabled
107 if (params.lineSearch.enabled)
108 {
109 double alpha = lineSearch<JacobianPolicy>(
110 fwdCorrs, revCorrs, pose, delta, current_cost,
111 rayDir, weighting, params.lineSearch, rotationScale);
112
113 if (params.verbose && alpha < 1.0)
114 {
115 std::cout << "\t\t\t line search: alpha=" << alpha << "\n";
116 }
117
118 delta *= alpha;
119 }
120
121 // Apply step
122 applyDeltaAndNormalize(pose, delta, rotationScale);
123
124 // Check convergence using separate translation/rotation thresholds
125 if (isDeltaConverged(delta, params.translationThreshold, params.rotationThreshold))
126 {
127 result.converged = true;
128 break;
129 }
130 }
131
132 result.pose = pose;
133 return result;
134}

149template<typename JacobianPolicy = RayJacobianSimplified>
150InnerSolveResult solveInner(
151 const std::vector<Correspondence>& fwdCorrs,
152 const std::vector<Correspondence>& revCorrs,
153 const Pose7& initialPose,
154 const Vector3& rayDir,
155 const GeometryWeighting& weighting = GeometryWeighting(),
156 const InnerParams& params = InnerParams())
157{
158 switch (params.solverType)
159 {
160 case SolverType::GaussNewton:
161 return solveInnerGN<JacobianPolicy>(fwdCorrs, revCorrs, initialPose, rayDir, weighting, params);
162 case SolverType::LevenbergMarquardt:
163 return solveInnerLM<JacobianPolicy>(fwdCorrs, revCorrs, initialPose, rayDir, weighting, params);
164 default:
165 return solveInnerGN<JacobianPolicy>(fwdCorrs, revCorrs, initialPose, rayDir, weighting, params);
166 }
167}

181template<typename JacobianPolicy = RayJacobianSimplified>
182ICPResult solveICP(
183 const Grid& source,
184 const Grid& target,
185 const Pose7& initialPose,
186 const Vector3& rayDir,
187 const GeometryWeighting& weighting = GeometryWeighting(),
188 const InnerParams& innerParams = InnerParams(),
189 const OuterParams& outerParams = OuterParams())
190{
191 ICPResult result;
192 result.pose = initialPose;
193 result.outer_iterations = 0;
194 result.total_inner_iterations = 0;
195 result.converged = false;
196
197 Pose7 pose = initialPose;
198 double prev_rms = std::numeric_limits<double>::max();
199
200 for (int outer = 0; outer < outerParams.maxIterations; ++outer)
201 {
202 result.outer_iterations++;
203
204 // Compute correspondences at current pose
205 Eigen::Isometry3d srcToTgt = pose7ToIsometry(pose);
206 auto corrs = computeBidirectionalCorrs(
207 source, target, rayDir.cast<float>(), srcToTgt, outerParams.maxDist);
208
209 if (outerParams.verbose)
210 {
211 std::cout << "\touter " << outer << ":\n";
212 std::cout << "\t\tfwd_corrs=" << corrs.forward.size()
213 << ", rev_corrs=" << corrs.reverse.size() << "\n";
214 }
215
216 // Inner loop with fixed correspondences
217 auto innerResult = solveInner<JacobianPolicy>(
218 corrs.forward, corrs.reverse, pose, rayDir, weighting, innerParams);
219
220 pose = innerResult.pose;
221 result.rms = innerResult.rms;
222 result.total_inner_iterations += innerResult.iterations;
223
224 if (outerParams.verbose)
225 {
226 std::cout << "\t\touter " << outer << ": rms=" << std::scientific
227 << std::setprecision(6) << result.rms << ", valid=" << innerResult.valid_count
228 << ", inner_iters=" << innerResult.iterations << "\n";
229 }
230
231 // Check convergence based on RMS change across outer iterations
232 double rms_change = std::abs(prev_rms - result.rms);
233 if (result.rms < outerParams.convergenceTol ||
234 rms_change < outerParams.convergenceTol * result.rms)
235 {
236 result.converged = true;
237 break;
238 }
239 prev_rms = result.rms;
240 }
241
242 if (outerParams.verbose)
243 {
244 std::cout << "Total inner iterations: " << result.total_inner_iterations << "\n";
245 }
246
247 result.pose = pose;
248 return result;
249}
