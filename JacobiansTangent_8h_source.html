68template<>
69class ForwardRayLeft<RayJacobianSimplified>
70{
71public:
72 using policy_tag = RayJacobianSimplified;
73
74 ForwardRayLeft(const Eigen::Vector3f& pS, const Eigen::Vector3f& qT,
75 const Eigen::Vector3f& nT, const Eigen::Vector3f& dT,
76 const GeometryWeighting& weighting = GeometryWeighting())
77 : pS_(pS), qT_(qT), nT_(nT), dT_(dT), weighting_(weighting)
78 {
79 }
80
89 double Evaluate(const Eigen::Matrix3d& R, const Eigen::Vector3d& t,
90 Eigen::Matrix<double, 1, 6>& J) const
91 {
92 // Transform source point to target frame
93 Eigen::Vector3d p = pS_.cast<double>();
94 Eigen::Vector3d x = R * p + t;
95
96 Eigen::Vector3d n = nT_.cast<double>();
97 Eigen::Vector3d q = qT_.cast<double>();
98 Eigen::Vector3d d = dT_.cast<double>();
99
100 // Ray normalization factor (denominator)
101 double b = n.dot(d);
102
103 // Compute weight based on incidence angle
104 double w = weighting_.weight(b);
105
106 // Numerator and residual
107 double a = n.dot(x - q);
108 double r = a / b;
109
110 // Jacobian: w * [+n^T | (x × n)^T] / b
111 // Simplified: ignores db/dphi
112 J.head<3>() = (w * n.transpose()) / b;
113 J.tail<3>() = (w * x.cross(n).transpose()) / b;
114
115 if (g_debugLeftJacobians && g_debugLeftCount < g_debugLeftMaxCount)
116 {
117 std::cout << " [FWD " << g_debugLeftCount << "] p=" << p.transpose()
118 << " -> x=" << x.transpose() << "\n";
119 std::cout << " q=" << q.transpose() << ", n=" << n.transpose() << "\n";
120 std::cout << " d=" << d.transpose() << ", b=" << b << ", w=" << w << "\n";
121 std::cout << " r=" << w * r << ", J=[" << J << "]\n";
122 g_debugLeftCount++;
123 }
124
125 return w * r;
126 }
127
131 double Residual(const Eigen::Matrix3d& R, const Eigen::Vector3d& t) const
132 {
133 Eigen::Vector3d p = pS_.cast<double>();
134 Eigen::Vector3d x = R * p + t;
135 Eigen::Vector3d n = nT_.cast<double>();
136 Eigen::Vector3d q = qT_.cast<double>();
137 Eigen::Vector3d d = dT_.cast<double>();
138
139 double b = n.dot(d);
140 double w = weighting_.weight(b);
141 double a = n.dot(x - q);
142
143 return w * a / b;
144 }
145
146private:
147 Eigen::Vector3f pS_, qT_, nT_, dT_;
148 GeometryWeighting weighting_;
149};

187template<>
188class ReverseRayLeft<RayJacobianSimplified>
189{
190public:
191 using policy_tag = RayJacobianSimplified;
192
193 ReverseRayLeft(const Eigen::Vector3f& pT, const Eigen::Vector3f& qS,
194 const Eigen::Vector3f& nS, const Eigen::Vector3f& dS,
195 const GeometryWeighting& weighting = GeometryWeighting())
196 : pT_(pT), qS_(qS), nS_(nS), dS_(dS), weighting_(weighting)
197 {
198 }
199
208 double Evaluate(const Eigen::Matrix3d& R, const Eigen::Vector3d& t,
209 Eigen::Matrix<double, 1, 6>& J) const
210 {
211 Eigen::Vector3d q = pT_.cast<double>(); // Ray origin in target-local
212 Eigen::Vector3d p = qS_.cast<double>(); // Hit point in source-local
213 Eigen::Vector3d ns = nS_.cast<double>(); // Normal in source-local
214 Eigen::Vector3d d = dS_.cast<double>();
215
216 // Inverse-transform q to source frame
217 Eigen::Vector3d xprime = R.transpose() * (q - t);
218
219 // Rotated normal (source normal in target frame)
220 Eigen::Vector3d nbar = R * ns;
221
222 // Ray normalization factor (in source frame)
223 double b = ns.dot(d);
224
225 // Compute weight based on incidence angle
226 double w = weighting_.weight(b);
227
228 // Residual in source frame
229 double a = ns.dot(xprime - p);
230 double r = a / b;
231
232 // Jacobian: w * [-nbar^T | (nbar × q)^T] / b
233 // Simplified: ignores db/dphi
234 J.head<3>() = (-w * nbar.transpose()) / b;
235 J.tail<3>() = (w * nbar.cross(q).transpose()) / b;
236
237 if (g_debugLeftJacobians && g_debugLeftCount < g_debugLeftMaxCount)
238 {
239 std::cout << " [REV " << g_debugLeftCount << "] q(tgt)=" << q.transpose()
240 << " -> x'=" << xprime.transpose() << "\n";
241 std::cout << " p(src)=" << p.transpose() << ", ns=" << ns.transpose() << "\n";
242 std::cout << " nbar=" << nbar.transpose() << ", b=" << b << ", w=" << w << "\n";
243 std::cout << " r=" << w * r << ", J=[" << J << "]\n";
244 g_debugLeftCount++;
245 }
246
247 return w * r;
248 }
249
253 double Residual(const Eigen::Matrix3d& R, const Eigen::Vector3d& t) const
254 {
255 Eigen::Vector3d q = pT_.cast<double>();
256 Eigen::Vector3d p = qS_.cast<double>();
257 Eigen::Vector3d ns = nS_.cast<double>();
258 Eigen::Vector3d d = dS_.cast<double>();
259
260 Eigen::Vector3d xprime = R.transpose() * (q - t);
261 double b = ns.dot(d);
262 double w = weighting_.weight(b);
263 double a = ns.dot(xprime - p);
264
265 return w * a / b;
266 }
267
268private:
269 Eigen::Vector3f pT_, qS_, nS_, dS_;
270 GeometryWeighting weighting_;
271};
