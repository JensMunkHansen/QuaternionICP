21std::vector<Edge> buildEdges(
22 const std::vector<Grid>& grids,
23 const std::vector<Pose7>& poses,
24 const OuterParams& outer,
25 bool verbose)
26{
27 std::vector<Edge> edges;
28 const size_t numGrids = grids.size();
29
30 if (numGrids < 2)
31 return edges;
32
33 // Compute world AABBs using current poses
34 std::vector<Grid::AABB> aabbs;
35 aabbs.reserve(numGrids);
36 for (size_t i = 0; i < numGrids; i++)
37 {
38 Eigen::Isometry3d worldPose = pose7ToIsometry(poses[i]);
39 aabbs.push_back(grids[i].computeWorldAABB(worldPose));
40 }
41
42 Eigen::Vector3f rayDirF = outer.rayDir.cast<float>();
43
44 // Generate candidate pairs (i, j) where i < j, respecting maxNeighbors limit
45 struct PairInfo
46 {
47 size_t i;
48 size_t j;
49 };
50 std::vector<PairInfo> pairs;
51 std::vector<int> neighborCount(numGrids, 0);
52
53 for (size_t i = 0; i < numGrids; i++)
54 {
55 for (size_t j = i + 1; j < numGrids; j++)
56 {
57 // Skip if either grid has reached neighbor limit
58 if (outer.maxNeighbors > 0 &&
59 (neighborCount[i] >= outer.maxNeighbors || neighborCount[j] >= outer.maxNeighbors))
60 {
61 continue;
62 }
63
64 if (aabbs[i].overlaps(aabbs[j], outer.maxDist))
65 {
66 pairs.push_back({i, j});
67 neighborCount[i]++;
68 neighborCount[j]++;
69 }
70 }
71 }
72
73 // Pre-allocate storage for parallel results (one per pair)
74 std::vector<BidirectionalCorrs> pairCorrs(pairs.size());
75
76 // Sort correspondences by quality: |nÂ·d| descending (higher = more perpendicular = better)
77 auto sortByQuality = [&rayDirF](std::vector<Correspondence>& corrs)
78 {
79 std::sort(corrs.begin(), corrs.end(),
80 [&rayDirF](const Correspondence& a, const Correspondence& b) {
81 return std::abs(a.tgtNormal.dot(rayDirF)) >
82 std::abs(b.tgtNormal.dot(rayDirF));
83 });
84 };
85
86 // Compute correspondences in parallel - each thread writes to its own slot
87 std::for_each(std::execution::par, pairs.begin(), pairs.end(),
88 [&](const PairInfo& pair)
89 {
90 size_t idx = static_cast<size_t>(&pair - pairs.data());
91
92 // Compute relative pose: srcToTgt = poses[j]^{-1} * poses[i]
93 Eigen::Isometry3d poseI = pose7ToIsometry(poses[pair.i]);
94 Eigen::Isometry3d poseJ = pose7ToIsometry(poses[pair.j]);
95 Eigen::Isometry3d srcToTgt = poseJ.inverse() * poseI;
96
97 pairCorrs[idx] = computeBidirectionalCorrs(
98 grids[pair.i], grids[pair.j], rayDirF, srcToTgt, outer.maxDist,
99 outer.subsampleX, outer.subsampleY);
100
101 // Sort by quality (best correspondences first)
102 sortByQuality(pairCorrs[idx].forward);
103 sortByQuality(pairCorrs[idx].reverse);
104 });
105
106 // Collect edges sequentially
107 for (size_t idx = 0; idx < pairs.size(); idx++)
108 {
109 const auto& pair = pairs[idx];
110 auto& corr = pairCorrs[idx];
111
112 // Apply maxCorrespondences limit (now keeps best ones due to sorting)
113 if (outer.maxCorrespondences > 0 &&
114 corr.forward.size() > static_cast<size_t>(outer.maxCorrespondences))
115 corr.forward.resize(outer.maxCorrespondences);
116 if (outer.maxCorrespondences > 0 &&
117 corr.reverse.size() > static_cast<size_t>(outer.maxCorrespondences))
118 corr.reverse.resize(outer.maxCorrespondences);
119
120 if (static_cast<int>(corr.forward.size()) >= outer.minMatch)
121 {
122 edges.push_back({static_cast<int>(pair.i), static_cast<int>(pair.j),
123 std::move(corr.forward)});
124 }
125
126 if (static_cast<int>(corr.reverse.size()) >= outer.minMatch)
127 {
128 edges.push_back({static_cast<int>(pair.j), static_cast<int>(pair.i),
129 std::move(corr.reverse)});
130 }
131
132 if (verbose && (!corr.forward.empty() || !corr.reverse.empty()))
133 {
134 std::cout << " Pair (" << pair.i << "," << pair.j << "): "
135 << corr.forward.size() << " fwd, "
136 << corr.reverse.size() << " rev\n";
137 }
138 }
139
140 return edges;
141}

143MultiViewICPResult runMultiViewICP(
144 std::vector<Grid>& grids,
145 const SessionParams& session,
146 const OuterParams& outer,
147 const InnerParams& inner)
148{
149 MultiViewICPResult result;
150 const size_t numGrids = grids.size();
151 const bool verbose = session.verbose || outer.verbose;
152
153 if (numGrids < 2)
154 {
155 std::cerr << "runMultiViewICP: need at least 2 grids\n";
156 return result;
157 }
158
159 // Convert grid.pose (Pose7) to Sophus::SE3d for Ceres optimization
160 std::vector<Sophus::SE3d> poses(numGrids);
161 for (size_t i = 0; i < numGrids; i++)
162 {
163 const auto& p = grids[i].pose;
164 Quaternion q(p[3], p[0], p[1], p[2]);
165 Vector3 t(p[4], p[5], p[6]);
166 poses[i] = Sophus::SE3d(q, t);
167 }
168
169 double cosAngleThresh = outer.weighting.enable_gate ? outer.weighting.tau : 0.0;
170 double prevRms = std::numeric_limits<double>::max();
171
172 // Outer loop
173 for (int outerIter = 0; outerIter < outer.maxIterations; outerIter++)
174 {
175 result.outerIterations++;
176
177 // Convert current Sophus poses to Pose7 for edge building
178 std::vector<Pose7> currentPoses(numGrids);
179 for (size_t i = 0; i < numGrids; i++)
180 {
181 Quaternion q = poses[i].unit_quaternion();
182 Vector3 t = poses[i].translation();
183 currentPoses[i] << q.x(), q.y(), q.z(), q.w(), t.x(), t.y(), t.z();
184 }
185
186 // Build edges using current poses
187 std::vector<Edge> edges = buildEdges(grids, currentPoses, outer, verbose);
188
189 size_t totalCorr = 0;
190 for (const auto& e : edges)
191 totalCorr += e.correspondences.size();
192
193 if (verbose)
194 {
195 std::cout << "Outer " << outerIter << ": " << edges.size() << " edges, "
196 << totalCorr << " correspondences\n";
197
198 // Print connectivity matrix on first iteration if small enough
199 if (outerIter == 0 && numGrids <= 20)
200 {
201 printConnectivityMatrix(buildCorrespondenceMatrix(edges, numGrids));
202 }
203 }
204
205 if (edges.empty())
206 {
207 std::cerr << "No edges found!\n";
208 break;
209 }
210
211 // Build Ceres problem
212 ceres::Problem problem;
213
214 // Add pose parameter blocks
215 std::vector<double*> poseData(numGrids);
216 for (size_t i = 0; i < numGrids; i++)
217 {
218 poseData[i] = poses[i].data();
219 problem.AddParameterBlock(poseData[i], Sophus::SE3d::num_parameters,
220 new SophusSE3Manifold());
221 }
222
223 // Fix first pose (gauge freedom)
224 if (session.fixFirstPose)
225 {
226 problem.SetParameterBlockConstant(poseData[0]);
227 }
228
229 // Add residuals for each edge
230 for (const auto& edge : edges)
231 {
232 // Determine if this is a forward edge (srcIdx < dstIdx) or reverse edge
233 bool isForward = edge.srcIdx < edge.dstIdx;
234 int poseIdxA = isForward ? edge.srcIdx : edge.dstIdx; // Canonical order: lower index
235 int poseIdxB = isForward ? edge.dstIdx : edge.srcIdx; // Canonical order: higher index
236
237 for (const auto& c : edge.correspondences)
238 {
239 double nDotD = c.tgtNormal.cast<double>().dot(outer.rayDir);
240 if (std::abs(nDotD) < cosAngleThresh)
241 continue;
242
243 ceres::CostFunction* costFn = nullptr;
244 if (isForward)
245 {
246 // Forward: ray from A to B, srcPoint in A, tgtPoint/tgtNormal in B
247 if (inner.jacobianPolicy == JacobianPolicy::Consistent)
248 {
249 costFn = ForwardRayCostTwoPose<RayJacobianConsistent>::Create(
250 c.srcPoint.cast<double>(),
251 c.tgtPoint.cast<double>(),
252 c.tgtNormal.cast<double>(),
253 outer.rayDir,
254 outer.weighting);
255 }
256 else
257 {
258 costFn = ForwardRayCostTwoPose<RayJacobianSimplified>::Create(
259 c.srcPoint.cast<double>(),
260 c.tgtPoint.cast<double>(),
261 c.tgtNormal.cast<double>(),
262 outer.rayDir,
263 outer.weighting);
264 }
265 }
266 else
267 {
268 // Reverse: ray from B to A, srcPoint in B (pB), tgtPoint/tgtNormal in A (qA, nA)
269 if (inner.jacobianPolicy == JacobianPolicy::Consistent)
270 {
271 costFn = ReverseRayCostTwoPose<RayJacobianConsistent>::Create(
272 c.srcPoint.cast<double>(), // pB
273 c.tgtPoint.cast<double>(), // qA
274 c.tgtNormal.cast<double>(), // nA
275 outer.rayDir,
276 outer.weighting);
277 }
278 else
279 {
280 costFn = ReverseRayCostTwoPose<RayJacobianSimplified>::Create(
281 c.srcPoint.cast<double>(), // pB
282 c.tgtPoint.cast<double>(), // qA
283 c.tgtNormal.cast<double>(), // nA
284 outer.rayDir,
285 outer.weighting);
286 }
287 }
288 problem.AddResidualBlock(costFn, nullptr, poseData[poseIdxA], poseData[poseIdxB]);
289 }
290 }
291
292 int validCount = problem.NumResidualBlocks();
293 if (validCount == 0)
294 {
295 std::cerr << "No valid residuals!\n";
296 break;
297 }
298
299 // Configure and solve using canonical params
300 ceres::Solver::Options options = toCeresSolverOptions(inner);
301
302 // Override linear solver for large multi-view problems (>2 grids)
303 // DenseQR is fine for 2 grids, but ITERATIVE_SCHUR scales better
304 if (inner.linearSolverType == LinearSolverType::DenseQR && numGrids > 2)
305 {
306 options.linear_solver_type = ceres::ITERATIVE_SCHUR;
307 // Preconditioner is already set by toCeresSolverOptions from inner.preconditionerType
308 }
309
310 ceres::Solver::Summary summary;
311 ceres::Solve(options, &problem, &summary);
312
313 result.totalInnerIterations += static_cast<int>(summary.iterations.size());
314
315 // Compute RMS
316 double rms = std::sqrt(2.0 * summary.final_cost / validCount);
317 result.rms = rms;
318
319 if (verbose)
320 {
321 std::cout << " RMS: " << std::scientific << std::setprecision(6) << rms
322 << ", valid: " << validCount
323 << ", ceres iters: " << summary.iterations.size() << "\n";
324 }
325
326 // Check convergence
327 double rmsChange = std::abs(prevRms - rms);
328 if (rms < outer.convergenceTol ||
329 rmsChange < outer.convergenceTol * rms)
330 {
331 result.converged = true;
332 break;
333 }
334 prevRms = rms;
335
336 // Early exit if Ceres converged in 1 iteration
337 if (summary.iterations.size() <= 1)
338 {
339 result.converged = true;
340 break;
341 }
342 }
343
344 // Write final poses back to grids
345 for (size_t i = 0; i < numGrids; i++)
346 {
347 Quaternion q = poses[i].unit_quaternion();
348 Vector3 t = poses[i].translation();
349 grids[i].pose << q.x(), q.y(), q.z(), q.w(), t.x(), t.y(), t.z();
350 }
351
352 return result;
353}

355std::vector<std::vector<int>> buildCorrespondenceMatrix(
356 const std::vector<Edge>& edges,
357 size_t numGrids)
358{
359 std::vector<std::vector<int>> matrix(numGrids, std::vector<int>(numGrids, 0));
360 for (const auto& edge : edges)
361 matrix[edge.srcIdx][edge.dstIdx] = static_cast<int>(edge.correspondences.size());
362 return matrix;
363}

365void printConnectivityMatrix(const std::vector<std::vector<int>>& counts)
366{
367 size_t n = counts.size();
368 if (n == 0)
369 return;
370
371 std::cout << "\nCorrespondence matrix (src -> dst):\n ";
372 for (size_t j = 0; j < n; j++)
373 std::cout << std::setw(6) << j;
374 std::cout << "\n";
375
376 for (size_t i = 0; i < n; i++)
377 {
378 std::cout << std::setw(4) << i << " ";
379 for (size_t j = 0; j < n; j++)
380 {
381 if (i == j)
382 std::cout << " -";
383 else if (counts[i][j] > 0)
384 std::cout << std::setw(6) << counts[i][j];
385 else
386 std::cout << " .";
387 }
388 std::cout << "\n";
389 }
390 std::cout << "\n";
391}

393void printResultsTable(
394 const std::vector<Pose7>& finalPoses,
395 const std::vector<Pose7>& initialPoses,
396 const std::vector<Pose7>& groundTruth)
397{
398 constexpr double RAD_TO_DEG = 180.0 / M_PI;
399 const size_t n = finalPoses.size();
400
401 std::cout << "\nPer-grid errors (vs ground truth):\n";
402 std::cout << "Grid | Init trans | Final trans | Init rot (deg) | Final rot (deg)\n";
403 std::cout << "-----|------------|-------------|----------------|----------------\n";
404
405 double initTotalTrans = 0.0, finalTotalTrans = 0.0;
406 double initTotalRot = 0.0, finalTotalRot = 0.0;
407
408 for (size_t i = 1; i < n; i++)
409 {
410 // Extract translations
411 Vector3 initT(initialPoses[i][4], initialPoses[i][5], initialPoses[i][6]);
412 Vector3 finalT(finalPoses[i][4], finalPoses[i][5], finalPoses[i][6]);
413 Vector3 gtT(groundTruth[i][4], groundTruth[i][5], groundTruth[i][6]);
414
415 double initTransErr = (initT - gtT).norm();
416 double finalTransErr = (finalT - gtT).norm();
417
418 // Extract quaternions and compute rotation errors
419 Quaternion initQ(initialPoses[i][3], initialPoses[i][0],
420 initialPoses[i][1], initialPoses[i][2]);
421 Quaternion finalQ(finalPoses[i][3], finalPoses[i][0],
422 finalPoses[i][1], finalPoses[i][2]);
423 Quaternion gtQ(groundTruth[i][3], groundTruth[i][0],
424 groundTruth[i][1], groundTruth[i][2]);
425
426 // Rotation error = angle of (q_est^-1 * q_gt)
427 Quaternion initRotDiff = initQ.conjugate() * gtQ;
428 Quaternion finalRotDiff = finalQ.conjugate() * gtQ;
429 double initRotErr = Eigen::AngleAxisd(initRotDiff).angle() * RAD_TO_DEG;
430 double finalRotErr = Eigen::AngleAxisd(finalRotDiff).angle() * RAD_TO_DEG;
431
432 std::cout << std::setw(4) << i << " | "
433 << std::setw(10) << std::fixed << std::setprecision(6) << initTransErr << " | "
434 << std::setw(11) << finalTransErr << " | "
435 << std::setw(14) << std::setprecision(4) << initRotErr << " | "
436 << std::setw(14) << finalRotErr << "\n";
437
438 initTotalTrans += initTransErr;
439 finalTotalTrans += finalTransErr;
440 initTotalRot += initRotErr;
441 finalTotalRot += finalRotErr;
442 }
443
444 std::cout << "-----|------------|-------------|----------------|----------------\n";
445 std::cout << " SUM | "
446 << std::setw(10) << std::fixed << std::setprecision(6) << initTotalTrans << " | "
447 << std::setw(11) << finalTotalTrans << " | "
448 << std::setw(14) << std::setprecision(4) << initTotalRot << " | "
449 << std::setw(14) << finalTotalRot << "\n";
450}
